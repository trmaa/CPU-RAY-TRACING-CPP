  5 #define MAP 0

  7 #include "camera.hpp"
  8 #include "scene.hpp"
  9 #include <SFML/Graphics.hpp>
 10 #include <SFML/Graphics/Color.hpp>
 11 #include <SFML/Window/Keyboard.hpp>
 12 #include <cmath>
 13 #include <glm/ext/vector_int2.hpp>
 14 #include <glm/geometric.hpp>
 15 #include <glm/glm.hpp>
 16 #include <string>
 17 #include <vector>
 18 #include <algorithm>
 19 #include <execution>
 20 
 21 sf::Color shader(int& x, int& y, glm::ivec2& buff_v, Camera& cam, Scene& scn, sf::Color& lastCol);
 22 
 23 class Window {
 24 private:
 25     glm::ivec2 m_viewport;
 26     sf::Image m_buffer;
 27     int m_frames;
 28     std::vector<glm::vec3> m_acumulation;
 29     sf::RenderWindow m_display;
 30 
 31     sf::Font m_font;
 32     sf::Text m_fpsText;
 33     uint16_t m_fps;
 34     uint8_t m_fpsLimit = 45;
 35 
 36     sf::Texture m_texture;
 37     sf::Sprite m_sprite;
 38 
 39     std::vector<int> x_values;
 40     std::vector<int> y_values;
 41 
 42 public:
 43     sf::Image& buffer() { return this->m_buffer; }
 44     sf::RenderWindow& display() { return this->m_display; }
 45 
 46     Window(int w, int h, std::string text)
 47         : m_viewport(w, h), m_frames(0), m_acumulation(w * h, glm::vec3(0.f, 0.f, 0.f)), x_values(w,0), y_values(h,0) {
 48         this->m_display.create(sf::VideoMode(w, h), text);
 49         this->m_buffer.create(w, h);
 50 
 51         if (!this->m_font.loadFromFile("./bin/fonts/pixelmix.ttf")) {
 52             std::cerr << ":) no font" << std::endl;
 53         }
 54 
 55         this->m_fpsText.setFont(this->m_font);
 56         this->m_fpsText.setCharacterSize(24);
 57         this->m_fpsText.setFillColor(sf::Color(0xff00ffff));
 58         this->m_fpsText.setPosition(10.f, 10.f);
 59 
 60         this->m_display.setFramerateLimit(this->m_fpsLimit);
 61 
 62         std::iota(y_values.begin(), y_values.end(), 0);
 63         std::iota(x_values.begin(), x_values.end(), 0);
 64     }
 65     ~Window() = default;
 66 
 67 public:
 68     void repaint(float& dt, Camera& cam, Scene& scn, sf::Event& ev) {
 69         this->m_fps = static_cast<uint16_t>(1.f / dt);
 70         this->m_fpsText.setString("FPS: " + std::to_string(this->m_fps));
 71 
 72         glm::ivec2 buff_v(this->m_buffer.getSize().x, this->m_buffer.getSize().y);
 73 
 74         bool resetAccumulation = false;
 75         if (ev.type == sf::Event::KeyPressed) {
 76             resetAccumulation = true;
 77             if (ev.key.code == sf::Keyboard::Tab) {
 78                 this->m_buffer.saveToFile("./bin/screenshot.jpg");
 79             }
 80         }
 81 
 82         if (resetAccumulation) {
 83             this->m_frames = 0;
 84             this->m_acumulation = std::vector<glm::vec3>(buff_v.x * buff_v.y, glm::vec3(0.f, 0.f, 0.f));
 85         }
 86         this->m_frames += 1;
 87 
 88         sf::Color lastCol;
 89         std::for_each(std::execution::par, y_values.begin(), y_values.end(), [&](int y) {
 90             std::for_each(std::execution::par, x_values.begin(), x_values.end(), [&](int x) {
 91                 const int index = buff_v.x * y + x;
 92                 sf::Color col = shader(x, y, buff_v, cam, scn, lastCol);
 93                 sf::Color fcolor;
 94 
 95                 //acumulation
 96                 this->m_acumulation[index] += glm::vec3(col.r, col.g, col.b);
 97 
 98                 fcolor = sf::Color(
 99                         this->m_acumulation[index].r/this->m_frames,
100                         this->m_acumulation[index].g/this->m_frames,
101                         this->m_acumulation[index].b/this->m_frames);
102 #if ANTIALIASING            
103                 this->m_buffer.setPixel(x, y, fcolor);
104 
105                 // antialiasing
106                 if (x < 1 || x > buff_v.x-1 || y < 1 || y > buff_v.y-1) {
107                     return;
108                 }
109 
110                 float r = 0.f, g = 0.f, b = 0.f;
111                 for (int offx = -1; offx <= 1; offx += 1) {
112                     for (int offy = -1; offy <= 1; offy += 1) {
113                         if(offx == 0 && offy == 0) {
114                             r += this->m_buffer.getPixel(x + offx, y + offy).r * 2;
115                             g += this->m_buffer.getPixel(x + offx, y + offy).g * 2;
116                             b += this->m_buffer.getPixel(x + offx, y + offy).b * 2;
117                         } else {
118                             r += this->m_buffer.getPixel(x + offx, y + offy).r;
119                             g += this->m_buffer.getPixel(x + offx, y + offy).g;
120                             b += this->m_buffer.getPixel(x + offx, y + offy).b;
121                         }
122                     }
123                 }
124                 r /= 10.0f;
125                 g /= 10.0f;
126                 b /= 10.0f;
127                 fcolor = sf::Color(static_cast<sf::Uint8>(r),
128                                    static_cast<sf::Uint8>(g),
129                                    static_cast<sf::Uint8>(b));
130 #endif
131                 this->m_buffer.setPixel(x, y, resetAccumulation ? col : fcolor);
132             });
133         });
134 
135         this->m_texture.loadFromImage(this->m_buffer);
136         this->m_sprite.setTexture(this->m_texture);
137         float scale = static_cast<float>(this->m_display.getSize().x) / buff_v.x;
138         this->m_sprite.setScale(scale, scale);
139 
140         this->m_display.clear();
141 
142         this->m_display.draw(this->m_sprite);
143         this->m_display.draw(this->m_fpsText);
144 
145 #if MAP
146         sf::CircleShape camera;
147         camera.setPosition(0.1f*cam.position().x+this->m_display.getSize().x/2,
148                 0.1f*cam.position().z+this->m_display.getSize().y/2);
149         camera.setFillColor(sf::Color(255, 255, 0));
150         camera.setRadius(10);
151         this->m_display.draw(camera);
152 
153         for (int i = 0; i < scn.triangle().size()+scn.sphere().size()-1; i++) {
154             sf::CircleShape plane;
155             plane.setPosition(0.1f*scn.object(i).center.x+this->m_display.getSize().x/2,
156                     0.1f*scn.object(i).center.z+this->m_display.getSize().y/2);
157             plane.setFillColor(sf::Color(255,0,255));
158             plane.setRadius(10);
159 
160             this->m_display.draw(plane);
161         }
162 #endif
163 
164         this->m_display.display();
165 
166         if (ev.key.code == sf::Keyboard::Tab) {
167             this->m_buffer.saveToFile("./bin/screenshot.jpg");
168         }
169     }
170 };

172 #endif
